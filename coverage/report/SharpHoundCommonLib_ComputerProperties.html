 383        /// <summary>
 384        ///     Attempts to parse all LDAP attributes outside of the ones already collected and converts them to a human
 385        ///     format using a best guess
 386        /// </summary>
 387        /// <param name="entry"></param>
 388        private static Dictionary<string, object> ParseAllProperties(ISearchResultEntry entry)
 389        {
 390            var flag = IsTextUnicodeFlags.IS_TEXT_UNICODE_STATISTICS;
 391            var props = new Dictionary<string, object>();
 392
 393            foreach (var property in entry.PropertyNames())
 394            {
 395                if (ReservedAttributes.Contains(property))
 396                    continue;
 397
 398                var collCount = entry.PropCount(property);
 399                if (collCount == 0)
 400                    continue;
 401
 402                if (collCount == 1)
 403                {
 404                    var testBytes = entry.GetByteProperty(property);
 405
 406                    if (testBytes == null || testBytes.Length == 0 ||
 407                        !IsTextUnicode(testBytes, testBytes.Length, ref flag)) continue;
 408
 409                    var testString = entry.GetProperty(property);
 410
 411                    if (!string.IsNullOrEmpty(testString))
 412                        if (property == "badpasswordtime")
 413                            props.Add(property, Helpers.ConvertFileTimeToUnixEpoch(testString));
 414                        else
 415                            props.Add(property, BestGuessConvert(testString));
 416                }
 417                else
 418                {
 419                    var arrBytes = entry.GetByteArrayProperty(property);
 420                    if (arrBytes.Length == 0 || !IsTextUnicode(arrBytes[0], arrBytes[0].Length, ref flag))
 421                        continue;
 422
 423                    var arr = entry.GetArrayProperty(property);
 424                    if (arr.Length > 0) props.Add(property, arr.Select(BestGuessConvert).ToArray());
 425                }
 426            }
 427
 428            return props;
 429        }

 431        /// <summary>
 432        ///     Does a best guess conversion of the property to a type useable by the UI
 433        /// </summary>
 434        /// <param name="property"></param>
 435        /// <returns></returns>
 436        private static object BestGuessConvert(string property)
 437        {
 438            //Parse boolean values
 439            if (bool.TryParse(property, out var boolResult)) return boolResult;
 440
 441            //A string ending with 0Z is likely a timestamp
 442            if (property.EndsWith("0Z")) return Helpers.ConvertTimestampToUnixEpoch(property);
 443
 444            //This string corresponds to the max int, and is usually set in accountexpires
 445            if (property == "9223372036854775807") return -1;
 446
 447            return property;
 448        }
